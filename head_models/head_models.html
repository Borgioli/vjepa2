Quickstart

Build the canonical phase RAG index

/path/to/data_root/vjepa2_1/.venv/bin/python \
  /path/to/data_root/vjepa2_1/phase_rag/phase_topk_embedding_index.py \
  --source-splits train \
  --top-k 8 \
  --batch-size 8 \
  --num-workers 4 \
  --device cuda \
  --skip-missing-clips \
  --output-dir /path/to/data_root/vjepa2_1/phase_rag_embeddings/native11_phase3_topk8_train75_NEW \
  --overwrite

Build all OvR phase head datasets + configs (reduced6)

/path/to/data_root/vjepa2_1/.venv/bin/python \
  /path/to/data_root/vjepa2_1/head_phases_ovr/build_phase_ovr_heads.py \
  --label-space reduced6 \
  --sources sitl surgenet

Preflight sanity checks

# 1) CSV paths actually resolve?
head -1 globus/surgenet_phases/yt_robotic_chole_native11_phases_val.csv \
  | awk '{print $1}' | xargs ls -lh

# 2) Index metadata says which backbone it expects?
cat vjepa2_1/phase_rag_embeddings/native11_phase3_topk8_train75/metadata.json | jq .

# 3) Backbone exists at the path the index expects?
ls -lh vjepa2_1/chekpoints_vjepar_surg_finetuned/single4_all_datasets_phase3_higher_res_g8_latest.pth.tar
